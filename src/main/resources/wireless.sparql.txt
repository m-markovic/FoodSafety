REGISTER QUERY noreasoning AS
PREFIX fs:<http://foodsafety-onto#>
SELECT Distinct ?type  ?time  ?value
FROM STREAM <http://foodsafety/wirelessTag> [RANGE 60s STEP 5s]
WHERE {
  fs:wirelesstag fs:observes ?obs.
  ?obs fs:value ?value.
  ?obs fs:time ?time.
  ?obs fs:type ?type
}
Order by DESC(?time)
